## Функция сохранения целостности БД

### Пример добавления новой строки с учетом внешних ключей

```sql
-- Добавление нового производителя и модели
INSERT INTO Производитель (Имя, ВебСайт) VALUES ('Toshiba', 'https://www.toshiba.com');
INSERT INTO Модель (ПроизводительID, НазваниеМодели, Объем, СкоростьШпинделя, Интерфейс)
VALUES (3, 'Toshiba X300', 2000, 7200, 'SATA');

-- Проверка, что при добавлении нового жесткого диска указана существующая модель
INSERT INTO ЖесткийДиск (СерийныйНомер, МодельID, ДатаПриобретения, ДатаВыходаИзСтроя, Комментарии)
VALUES ('SN345678', 3, '2023-03-10', NULL, 'New disk added');
```

### Пример изменения данных с учетом связей

```sql
-- Изменение данных в таблице ЖесткийДиск
UPDATE ЖесткийДиск
SET ДатаВыходаИзСтроя = '2024-05-10'
WHERE СерийныйНомер = 'SN123456';
```

### Пример удаления данных

```sql
-- Удаление данных из таблицы ЖесткийДиск
DELETE FROM ЖесткийДиск
WHERE СерийныйНомер = 'SN789012';
```

### Демонстрация ссылочной целостности при удалении данных

```sql
-- Попытка удаления производителя, связанного с моделью
DELETE FROM Производитель
WHERE ПроизводительID = 1; -- Будет ошибка, так как есть связанные модели

-- Правильное удаление с учетом зависимостей
DELETE FROM ЖесткийДиск
WHERE МодельID = 1;

DELETE FROM Модель
WHERE ПроизводительID = 1;

DELETE FROM Производитель
WHERE ПроизводительID = 1;
```
